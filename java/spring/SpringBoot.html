    public int search( List<Integer> nums, int target) {
        int low = 0;
        int high = nums.size()-1;
        int mid = 0;
        while (low <= high) {
            mid = low + (high - low) / 2;
            if (nums.get(mid) > target)
                high = mid - 1;
            else if (nums.get(mid) < target)
                low = mid + 1;
            else
                return nums.get(mid);
        }
        return nums.get(mid+1);
    }
